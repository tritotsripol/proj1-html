    Options

    <input type="(checkbox or radio)" id="option" name="option" />
    <label for="option">(Text)</label>
    ...(depends on the number of options)

    Multiple option: "checkbox"
    Single option: "radio"
    Example Output
    Set to default

    <input type="checkbox" id="option" name="option" value="(Variable)"/>

    ...(depends on the number of options)

    </select>
